# main fuction


## classic main func

'''
int main(void);
int main(int args, char ** argv);
int main(int args, char *argv[]);

'''
## main func with unicode character

'''
int wmain(void);
int wmain(int args, wchar_t **argv);
int wmain(int args, wchar_t *argv[]);
'''
1. wchar_t -> new datatype -> ref.
2. wmain -> wide main
